template<typename T>
class pk::detail::arrayptr< T >

value 타입. 1차원 배열 포인터를 감싸는 타입으로 set_func, bind 콜백의 매개변수에서만 사용한다. 스크립트에서 지원하는 array 타입으로 변환하기 귀찮아서 만든 타입이다.

int arr[3];
asIScriptEngine* engine;
asIScriptContext* context;
// 불필요한 메모리 할당 발생. 타입까지 지정해야 하는 번거로움.
CScriptArray* arr2 = CScriptArray::Create(engine->GetTypeInfoByDecl("array<int>"), 3);
for (int i = 0; i < 3; i++)
arr2->SetValue(i, &arr[i]);
context->SetArgObject(0, arr2);
context->Execute();
arr2->Release();
// 매우 간단!
arrayptr arr3 = arrayptr(arr, sizeof(int), 3);
context->SetArgObject(0, &arr3);
context->Execute();
